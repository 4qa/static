Configuration

// build.gradle.kts
import org.jetbrains.dokka.auxiliaryDocs.*

plugins {
    id("org.jetbrains.kotlin.jvm")
    id("org.jetbrains.dokka")
}

buildscript {
    dependencies {
        classpath("org.jetbrains.dokka:dokka-base:$dokka_version")
        classpath("org.jetbrains.dokka:auxiliary-docs:$dokka_version")
    }
}

dependencies {
    dokkaHtmlPlugin("org.jetbrains.dokka:auxiliary-docs:$dokka_version")
}


tasks.dokkaHtml.configure{
    pluginConfiguration<AuxiliaryDocsPlugin, AuxiliaryConfiguration> {
        nodesDir = File("docs") // dir with a plain list of md files
        entryPointNode = File("docs/Index.md") // location of the desired root page
        apiReferenceNodeName = "API Reference" // name of the node with generated API documentation
    }
}
